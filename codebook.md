#Variables Description for the tidydata.txt dataset

"subject.id"
	 uniquely identifies the subject participating in the experiment
	 takes values 1 to 30
	 
"activity"
	The activity during which the measurements were taken
	takes values:
	
	WALKING
	WALKING_UPSTAIRS
	WALKING_DOWNSTAIRS
	SITTING
	STANDING
	LAYING
